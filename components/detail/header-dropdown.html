var handleHideMenu, hideMenu, initMenu, showMenu, touchEndHandler, touchStartHandler;

window.touchesInAction = {};

touchStartHandler = function(event) {
  var i, len, results, t, touches;
  touches = event.changedTouches;
  results = [];
  for (i = 0, len = touches.length; i < len; i++) {
    t = touches[i];
    results.push(window.touchesInAction["$" + t.identifier] = {
      identifier: t.identifier,
      pageX: t.pageX,
      pageY: t.pageY
    });
  }
  return results;
};

touchEndHandler = function(cb, event) {
  var i, len, t, theTouchInfo, touches;
  touches = event.changedTouches;
  for (i = 0, len = touches.length; i < len; i++) {
    t = touches[i];
    theTouchInfo = window.touchesInAction["$" + t.identifier];
    theTouchInfo.dx = t.pageX - theTouchInfo.pageX;
    theTouchInfo.dy = t.pageY - theTouchInfo.pageY;
  }
  if (theTouchInfo.dx === 0 && theTouchInfo.dy === 0) {
    return cb(event);
  }
};

window.menuOpen = false;

showMenu = function(e) {
  var menu;
  if (window.menuOpen) {
    return;
  }
  e.preventDefault();
  menu = document.getElementById('website-menu');
  return Velocity(document.getElementById('menu-container'), {
    left: 0,
    opacity: 1
  }, {
    display: 'block',
    duration: 250,
    complete: function() {
      menu.className = "[ header__menu ] [ menu menu--open ]";
      document.body.className = "no-scroll";
      return window.menuOpen = true;
    }
  });
};

handleHideMenu = function(e) {
  if (!window.menuOpen) {
    return;
  }
  e.preventDefault();
  if (e.target.tagName !== 'A') {
    if (e.target.id === 'home-menu-btn') {
      window.location = '/';
      return;
    }
    return hideMenu();
  } else {
    return setTimeout(function() {
      return window.location = e.target.href;
    }, 100);
  }
};

hideMenu = function() {
  var menu;
  menu = document.getElementById('website-menu');
  return Velocity(document.getElementById('menu-container'), {
    left: '100%',
    opacity: 0
  }, {
    display: 'none',
    duration: 250,
    complete: function() {
      menu.className = "[ header__menu ] [ menu ]";
      document.body.className = "";
      return window.menuOpen = false;
    }
  });
};

initMenu = function() {
  var hideMenuBtn, showMenuBtn;
  showMenuBtn = document.getElementById('show-menu');
  showMenuBtn.addEventListener('touchstart', showMenu);
  showMenuBtn.addEventListener('click', showMenu);
  hideMenuBtn = document.getElementById('menu-container');
  hideMenuBtn.addEventListener('click', handleHideMenu);
  hideMenuBtn.addEventListener('touchstart', touchStartHandler);
  return hideMenuBtn.addEventListener('touchend', touchEndHandler.bind(this, handleHideMenu));
};

initMenu();
</script>
